7// cxDate.h - Date class for the cxWidgets library (C++14-compatible).
8// This is a modernized version of the original 'date' class (date.h/date.cpp),
9// renamed to cxDate to follow cxWidgets naming conventions.
10//
11// Changes from the original 'date' class:
12// - Class renamed from 'date' to 'cxDate'
13// - noexcept specifications added to non-throwing methods
14// - std::chrono used in today() for current date retrieval
15// - std::stoi() replaces atoi() for safer string-to-int conversion
16// - std::to_string() replaces the custom anythingToString<T>() template
17// - static cxDate today() added as a clear, expressive convenience method
18// - Default member initializers used in the class definition
19// - = default used for copy constructor, copy assignment, and destructor
